<script>
function editar() {
    document.getElementById("mensaje").contentEditable = true;
    document.getElementById("mensaje").focus();
}

function guardar() {
    const texto = document.getElementById("mensaje").innerHTML;
    localStorage.setItem("mensajePremium", texto);
    document.getElementById("mensaje").contentEditable = false;
    alert("Mensaje guardado");
}

window.onload = function() {
    const guardado = localStorage.getItem("mensajePremium");
    if (guardado) {
        document.getElementById("mensaje").innerHTML = guardado;
    }
}
